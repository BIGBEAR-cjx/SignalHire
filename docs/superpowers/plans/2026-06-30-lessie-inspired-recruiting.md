# Lessie-Inspired Recruiting Module Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the prioritized Lessie-inspired recruiting improvements without diluting SignalHire's evidence-first positioning: fast unverified lead preview, clearer source mix UX, and stronger outreach sequence workflow.

**Architecture:** Keep the existing CandidateGraph, contact resolution, Gmail outreach, and project workspace as the base. Add thin view-model modules and focused UI panels rather than new persistence unless a task explicitly needs it. Controller agent coordinates parallel work; implementer subagents work in isolated task boundaries; final acceptance is performed by an independent agent that did not implement the changes.

**Tech Stack:** Next.js App Router, React, TypeScript, Node ESM modules, Insforge-backed tables, existing `node:test` suites, `npm --prefix web run build`.

## Controller And Subagent Operating Model

Controller agent responsibilities:

- Own this plan file and the PRD.
- Dispatch one fresh implementer subagent per task.
- Prevent overlapping edits to the same file in the same wave.
- After each task, dispatch a spec review agent, then a code quality review agent.
- Only mark a task complete after both reviews pass.
- Dispatch the independent acceptance agent after all implementation tasks complete.

Required subagent roles:

- `Implementer Agent`: writes tests first, implements the task, runs task-level verification, reports `DONE`, `DONE_WITH_CONCERNS`, `NEEDS_CONTEXT`, or `BLOCKED`.
- `Spec Review Agent`: checks only PRD/plan compliance.
- `Code Quality Review Agent`: checks bugs, regression risk, overengineering, accessibility, and missing tests.
- `Independent Acceptance Agent`: performs final product-level verification from user workflows and must not modify code unless the controller explicitly assigns a follow-up fix.

Parallel waves:

- Wave 0: Task 1 only.
- Wave 1 parallel: Task 2 and Task 3, but Task 3 must not edit search-result UI until Task 2 files are known.
- Wave 2: Task 4.
- Wave 3: Task 5 independent acceptance.

```js
function isRecord(value) {
  return Boolean(value && typeof value === "object" && !Array.isArray(value));
}

function cleanString(value) {
  return String(value ?? "").replace(/\s+/g, " ").trim();
}

function cleanKey(value) {
  return cleanString(value).toLowerCase().replace(/^https?:\/\//, "").replace(/^www\./, "").replace(/\/+$/, "");
}

function arrayOf(value) {
  return Array.isArray(value) ? value : [];
}

function eventRows(run) {
  const progress = isRecord(run?.progress) ? run.progress : {};
  const execution = isRecord(progress.agent_execution) ? progress.agent_execution : {};
  return arrayOf(execution.candidate_submission_events);
}

function openEvidenceRows(rows) {
  return arrayOf(rows).map((row) => {
    const source = isRecord(row) ? row : {};
    return {
      name: source.title || source.name || source.candidate_name,
      headline: source.snippet,
      company: "",
      source_type: source.source_type || source.provider || "open_evidence",
      source_url: source.source_url || source.url,
      match_reason: source.snippet || source.reason,
      confidence: source.confidence || "low",
    };
  });
}

function resultHasVerifiedCandidates(run) {
  const result = isRecord(run?.result) ? run.result : {};
  return arrayOf(result.shortlist).length > 0 || arrayOf(result.candidates).length > 0;
}

function normalizeLead(row) {
  const source = isRecord(row) ? row : {};
  return {
    id: cleanKey(source.source_url) || cleanKey(`${source.name}:${source.company || source.current_company}`),
    label: "unverified lead",
    candidate_name: cleanString(source.name || source.candidate_name),
    headline: cleanString(source.headline || source.current_role || source.title),
    company: cleanString(source.company || source.current_company),
    source_type: cleanString(source.source_type || source.provider || "unknown"),
    source_url: cleanString(source.source_url || source.url),
    possible_match_reason: cleanString(source.match_reason || source.reason || source.summary),
    missing_evidence: ["public evidence packet", "contact provenance"],
    next_verification_step: "Run deep research before recommendation or outreach.",
    confidence: cleanString(source.confidence || "low"),
    feedback_state: "untouched",
    can_outreach: false,
  };
}

export function buildLeadPreviewView({ run = {}, openEvidenceLeads = [] } = {}) {
  if (resultHasVerifiedCandidates(run)) {
    return { status: "verified_results_available", items: [], feedback_constraints: [] };
  }
  const seen = new Set();
  const items = [];
  for (const row of [...eventRows(run), ...openEvidenceRows(openEvidenceLeads)]) {
    const item = normalizeLead(row);
    if (!item.id || seen.has(item.id)) continue;
    seen.add(item.id);
    items.push(item);
  }
  return {
    status: items.length > 0 ? "preview_available" : "waiting_for_leads",
    items: items.slice(0, 12),
    feedback_constraints: [],
  };
}
```

```js
function cleanString(value) {
  return String(value ?? "").trim();
}

function hostOf(url) {
  try {
    return new URL(cleanString(url)).hostname.replace(/^www\./, "");
  } catch {
    return "";
  }
}

export function classifySourceType(source = {}) {
  const sourceType = cleanString(source.source_type);
  const provider = cleanString(source.provider).toLowerCase();
  const url = cleanString(source.source_url || source.url);
  const host = hostOf(url);
  const text = `${sourceType} ${provider} ${url} ${cleanString(source.source_family)} ${cleanString(source.title)} ${cleanString(source.snippet)}`.toLowerCase();
  if (host === "github.com" || text.includes("github")) return "github";
  if (/semanticscholar\.org|openreview\.net|arxiv\.org|doi\.org/.test(host) || /semantic scholar|openalex|openreview|paper|publication/.test(text)) return "paper";
  if (/\/(team|about|people|company|jobs|profile)(\/|$)/i.test(url) || /company page|team page|employment/.test(text)) return "company_page";
  if (sourceType === "people_api" || provider.includes("hunter") || provider.includes("pdl")) return "people_api";
```

```js
export function buildRoleOutreachSettings(source = {}) {

export function canAutoSendFollowUp({ settings = {}, message = {}, thread = {} } = {}) {
  const normalized = buildRoleOutreachSettings(settings);
  if (!normalized.auto_follow_up_only) return false;
  if (message.step === 1) return false;
  if (message.approved !== true) return false;
  if (["stopped", "replied", "bounced", "not_interested"].includes(String(thread.status))) return false;
  return message.send_mode === "draft_for_review";
}
```

```js
function cleanString(value) {
  return String(value ?? "").replace(/\s+/g, " ").trim();
}

function primaryEmailSummary(profile = {}) {
  const email = Array.isArray(profile.emails) ? profile.emails[0] : null;
  if (!email) return "No sourced contact yet";
  return `${cleanString(email.source) || "sourced contact"} / ${cleanString(email.confidence) || "unknown confidence"}`;
}

export function buildAgencyOutreachActivityDigest({ roleName = "", threads = [] } = {}) {
  const lines = [`Outreach activity digest: ${cleanString(roleName) || "Role"}`];
  for (const thread of Array.isArray(threads) ? threads : []) {
    lines.push([
      `Candidate: ${cleanString(thread.candidate_name) || "Unknown"}`,
      `Status: ${cleanString(thread.status) || "draft"}`,
      `Next follow-up: ${cleanString(thread.next_follow_up_at) || "not scheduled"}`,
      `Evidence angle: ${cleanString(thread.evidence_angle) || cleanString(thread.contact_angle) || "not specified"}`,
      `Contact provenance: ${primaryEmailSummary(thread.contact_profile || {})}`,
    ].join("\n"));
  }
  return lines.join("\n\n");
}
```
